fetch('https://jsonplaceholder.typicode.com/todos') .then(response => response.json()) .then(json => { let div_list = []; for (let i = 0; i < json.length; i++) { let div = document.createElement("div"); div.textContent = json[i].title; div_list.push(div); } for (let i = 0; i < div_list.length; i++) { document.body.appendChild(div_list[i]); } })
